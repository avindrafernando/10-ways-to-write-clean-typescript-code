What is Clean Code?

  • Easy to understand
    • The execution flow of the entire application
    • How the different objects collaborate with each other
    • The role and responsibility of each component
    • What each method/function does
    • What is the purpose of each expression and variable

Generic Arguments

‚ùå any for generic arguments

// NOT SO CLEAN
function rearrange(list: any, startIndex: number, endIndex: number): any {
  ...
  return rearrangedList
}


// myStrings is of type 'any'
const myStrings = rearrange(["a", "b", "c"], 0, 1)

// myNumbers is of type 'any'
const myNumbers = rearrange([1, 2, 3], 0, 1)

‚úÖ Use a generic type

// MUCH BETTER
function rearrange<T>(list: T[], startIndex: number, endIndex: number): T[] {
  ...
  return rearrangedList
}



// myStrings is of type 'string[]'
const myStrings = rearrange(["a", "b", "c"], 0, 1)

// myNumbers is of type 'number[]'
const myNumbers = rearrange([1, 2, 3], 0, 1)

Minimize Duplication

‚ùå Duplicated types

// NOT SO CLEAN 
type Product = {
  id: string
  sku?: string
  upc?: string
}

// The id and the sku are now required
type ProductWithSku = {
  id: string
  sku: string
  upc?: string
}

‚úÖ Mapped types

// MUCH BETTER
type WithRequiredProperty<Type, Key extends keyof Type> = Type & {
  [Property in Key]-?: Type[Property]
}
 
type Product = {
  id: string
  sku?: string
  upc?: string
}

// The id and the sku are now required
type ProductWithSku = WithRequiredProperty<Product, 'sku'>

// The Required utility type makes all properties required
type ProductWithSkuAndUpc = Required<Product>

‚ùå Duplicated types

// NOT SO CLEAN 
type Product = {
  id: string
  sku?: string
  upc?: string
}

// Product without sku
type ProductWithoutSku = {
  id: string
  upc?: string
}

‚úÖ Mapped types

// MUCH BETTER
type RemoveSkuField<Type> = {
    [Property in keyof Type as Exclude<Property, 'sku'>]: Type[Property]
};

type Product = {
  id: string
  sku?: string
  upc?: string
}


type ProductWithoutSku = RemoveSkuField<Product>

// EVEN BETTER - Using Omit utility type
type ProductWithoutSku = Omit<Product, 'sku'>

¬†‚ÑπÔ∏è Mapped types are great for creating a type based on another type.

Callbacks

‚ùå Using any on callback returning nothing

// NOT SO CLEAN 
function doWork(doMoreWork: () => any) {
  doMoreWork()
}

‚úÖ Callback returning void

// MUCH BETTER
function doWork(doMoreWork: () => void) {
  doMoreWork()
}

// Property 'doAdditionalWork' does not exist on type 'void'.(2339)
function doWork(doMoreWork: () => void) {
  const result = doMoreWork()
  result.doAdditionalWork()
}

‚ùå Using Function on callback returning a value

// NOT SO CLEAN 
function doWork(resultOfMoreWork: Function) {
  return resultOfMoreWork()
}

‚úÖ Callback returning a specific type

// MUCH BETTER
function doWork(resultOfMoreWork: (item: number) => number) {
  return resultOfMoreWork()
}

‚úÖ Type predicates

// MUCH BETTER
type TextChoice = {
  type: 'text'
  placeholder: string
}

type RadioChoice = {
  type: 'radio'
  options: string[]
}

type Choice = TextChoice | RadioChoice

function isTextChoice(choice: Choice): choice is TextChoice {
  return (choice as TextChoice).placeholder !== undefined
}

function logChoiceToConsole(choice: Choice): void {

    console.log('I am a text choice with a placeholder.', choice.placeholder)
  } else {
    console.log('I am a radio choice.')
  }
}

Filtering undefined

‚ùå Using Boolean

// NOT SO CLEAN 
const filteredFruits = ['apple', 'orange', undefined, 'kiwi'].filter(Boolean)

[LOG]: ["apple", "orange", "kiwi"]
// const filteredFruits: (string | undefined)[]

‚úÖ Type guard with type predicate

// MUCH BETTER
function isDefined<T>(listItem: T | undefined): listItem is T {
  if (listItem === undefined) {
    return false
  }
  return true
}

const filteredFruits = ['apple', 'orange', undefined, 'kiwi'].filter(isDefined)

// const filteredFruits: string[]

type TypeA = string
type TypeB = number

        return "23" as TypeA
    }
    return 23 as TypeB

¬†Extract Keys of a Value Type

ü§î

type Address = { 
  streetName: string;
  city: string;
  houseNumber: number;
  zipCode: number;
}

// type StringKeysOfAdresss = "streetName" | "city"

üôÇ Types to extract keys

// WORKS BUT CAN BE CLEANER
type MatchingStringKeys<Type> = {
  [Key in keyof Type]: Type[Key] extends string ? Key : never
}[keyof Type]
 
// type StringKeysOfAdresss = "streetName" | "city"
type StringKeysOfAdresss = MatchingStringKeys<Address>

type MatchingNumberKeys<Type> = {
  [Key in keyof Type]: Type[Key] extends number ? Key : never
}[keyof Type]

// type NumberKeysOfAdresss = "houseNumber" | "zipCode"
type NumberKeysOfAdresss = MatchingNumberKeys<Address>

üòÅ Let's make it generic

// MUCH BETTER
type MatchingKeys<Type, FilterType> = {
  [Key in keyof Type]: Type[Key] extends FilterType
    ? Key
    : never
}[keyof Type]
 
// type StringKeysOfAdresss = "streetName" | "city"
type StringKeysOfAdresss = MatchingKeys<Address, string>

// type NumberKeysOfAdresss = "houseNumber" | "zipCode"
type NumberKeysOfAdresss = MatchingKeys<Address, number>
