Clean Code

Clean code is code that is easy to understand and easy to change.

What is Clean Code?

  • Easy to understand
    • The execution flow of the entire application
    • How the different objects collaborate with each other
    • The role and responsibility of each component
    • What each method/function does
    • What is the purpose of each expression and variable

Ways

Strict Mode

❌ Strict mode disabled

// NOT SO CLEAN
const myFruit = ['apple', 'orange', 'kiwi'].find((fruit) => fruit === 'banana')

const eatFruit = (fruit: string) => {
  //
}

// Argument of type 'string | undefined' is not assignable to parameter of type 'string'.
//  Type 'undefined' is not assignable to type 'string'.(2345)
eatFruit(myFruit)

tsconfig.json ➡️ "strict": true

  • alwaysStrict
  • strictBindCallApply
  • strictFunctionTypes
  • strictNullChecks
  • strictPropertyInitialization
  • noImplicitAny
  • noImplicitThis
  • useUnknownInCatchVariables

✅ My 5 step approach

1️⃣ Enable strict mode.
2️⃣ Disable all strict rules except one.
3️⃣ Fix all errors related to the enabled rule.
4️⃣ Enable another rule.
5️⃣ Rinse and repeat.

Read Only

⚠️ Object.freeze

// NOT SO CLEAN
const product = Object.freeze({
  name: 'Candy',
  codes: {
    sku: '010000',
    upc: '72527273070'
  }
}) 

// Cannot assign to 'name' because it is a read-only property.(2540)
product.name = 'Soda'

// No Error
product.codes.sku = '010002'

✅ as const

// MUCH BETTER
const product = {
  name: 'Chocolate',
  codes: {
    sku: '010001',
    upc: '72527273072'
  }
} as const  

// Cannot assign to 'name' because it is a read-only property.(2540)
product.name = 'Bread'

// Cannot assign to 'sku' because it is a read-only property.(2540)
product.codes.sku = '010003'

But,
Caution ⚠️

 ⚠️ Object.freeze is a JavaScript construct which provides enforcement at run-time.

Generic Arguments

❌ any for generic arguments

// NOT SO CLEAN
function rearrange(list: any, startIndex: number, endIndex: number): any {
  ...
  return rearrangedList
}


// myStrings is of type 'any'
const myStrings = rearrange(["a", "b", "c"], 0, 1)

// myNumbers is of type 'any'
const myNumbers = rearrange([1, 2, 3], 0, 1)

✅ Use a generic type

// MUCH BETTER
function rearrange<T>(list: T[], startIndex: number, endIndex: number): T[] {
  ...
  return rearrangedList
}



// myStrings is of type 'string[]'
const myStrings = rearrange(["a", "b", "c"], 0, 1)

// myNumbers is of type 'number[]'
const myNumbers = rearrange([1, 2, 3], 0, 1)

 ℹ️ Generic types preserve type information.

Minimize Duplication

❌ Duplicated types

// NOT SO CLEAN 
type Product = {
  id: string
  sku?: string
  upc?: string
}

// The id and the sku are now required
type ProductWithSku = {
  id: string
  sku: string
  upc?: string
}

✅ Mapped types

// MUCH BETTER
type WithRequiredProperty<Type, Key extends keyof Type> = Type & {
  [Property in Key]-?: Type[Property]
}
 
type Product = {
  id: string
  sku?: string
  upc?: string
}

// The id and the sku are now required
type ProductWithSku = WithRequiredProperty<Product, 'sku'>

// The Required utility type makes all properties required
type ProductWithSkuAndUpc = Required<Product>

❌ Duplicated types

// NOT SO CLEAN 
type Product = {
  id: string
  sku?: string
  upc?: string
}

// Product without sku
type ProductWithoutSku = {
  id: string
  upc?: string
}

✅ Mapped types

// MUCH BETTER
type RemoveSkuField<Type> = {
    [Property in keyof Type as Exclude<Property, 'sku'>]: Type[Property]
};

type Product = {
  id: string
  sku?: string
  upc?: string
}


type ProductWithoutSku = RemoveSkuField<Product>

// EVEN BETTER - Using Omit utility type
type ProductWithoutSku = Omit<Product, 'sku'>

 ℹ️ Mapped types are great for creating a type based on another type.

Callbacks

❌ Using any on callback returning nothing

// NOT SO CLEAN 
function doWork(doMoreWork: () => any) {
  doMoreWork()
}

✅ Callback returning void

// MUCH BETTER
function doWork(doMoreWork: () => void) {
  doMoreWork()
}

// Property 'doAdditionalWork' does not exist on type 'void'.(2339)
function doWork(doMoreWork: () => void) {
  const result = doMoreWork()
  result.doAdditionalWork()
}

❌ Using Function on callback returning a value

// NOT SO CLEAN 
function doWork(resultOfMoreWork: Function) {
  return resultOfMoreWork()
}

✅ Callback returning a specific type

// MUCH BETTER
function doWork(resultOfMoreWork: (item: number) => number) {
  return resultOfMoreWork()
}

Narrowing Types

🤔

// NOT SO CLEAN 
type TextChoice = {
  type: 'text'
  placeholder: string
}

type RadioChoice = {
  type: 'radio'
  options: string[]
}

type Choice = TextChoice | RadioChoice

function logChoiceToConsole(choice: Choice): void {
//Property 'placeholder' does not exist on type 'Choice'.
//  Property 'placeholder' does not exist on type 'RadioChoice'.
  if (choice.placeholder) {
    console.log('I am a text choice with a placeholder.', choice.placeholder)
  } else {
    console.log('I am a radio choice.')
  }
}

❌ Inline check using in operator

// NOT SO CLEAN 
type TextChoice = {
  type: 'text'
  placeholder: string
}

type RadioChoice = {
  type: 'radio'
  options: string[]
}

type Choice = TextChoice | RadioChoice

function logChoiceToConsole(choice: Choice): void {
  if ('placeholder' in choice) {
    console.log('I am a text choice with a placeholder.', choice.placeholder)
  } else {
    console.log('I am a radio choice.')
  }
}

✅ Type predicates

// MUCH BETTER
type TextChoice = {
  type: 'text'
  placeholder: string
}

type RadioChoice = {
  type: 'radio'
  options: string[]
}

type Choice = TextChoice | RadioChoice

function isTextChoice(choice: Choice): choice is TextChoice {
  return (choice as TextChoice).placeholder !== undefined
}

function logChoiceToConsole(choice: Choice): void {
  if (isTextChoice(choice)) {
    console.log('I am a text choice with a placeholder.', choice.placeholder)
  } else {
    console.log('I am a radio choice.')
  }
}

➕ Let's add one more type

type CheckboxChoice = {
  type: 'checkbox',
  options: string[]
}

type Choice = TextChoice | RadioChoice | CheckboxChoice

function isTextChoice(choice: Choice): choice is TextChoice {
  return (choice as TextChoice).placeholder !== undefined
}

function isRadioChoice(choice: Choice): choice is RadioChoice {
  return (choice as RadioChoice).options !== undefined
}

function logChoiceToConsole(choice: Choice): void {
  if (isTextChoice(choice)) {
    console.log('I am a text choice with a placeholder.', choice.placeholder)
  } else if (isRadioChoice(choice)) {
    console.log('I am a radio choice.')
  } else {
    console.log('I am a checkbox choice.')
  }
}

✅ Discriminated unions

type TextChoice = {
  type: 'text'
  placeholder: string
}

type RadioChoice = {
  type: 'radio'
  options: string[]
}

type CheckboxChoice = {
  type: 'checkbox',
  options: string[]
}

type Choice = TextChoice | RadioChoice | CheckboxChoice

function logChoiceToConsole(choice: Choice): void {
  if (choice.type === 'text') {
    console.log('I am a text choice with a placeholder.', choice.placeholder)
  } else if (choice.type === 'radio') {
    console.log('I am a radio choice.')
  } else {
    console.log('I am a checkbox choice.')
  }
}

Types with Many Shapes

❌ Lots of optionals

// NOT SO CLEAN 
type Choice = {
  type: 'text' | 'radio'
  placeholder?: string
  options?: string[]
}

// ❌ - We shouldn't be allowed to decalare options on a Text Choice
const myTextChoice: Choice = {
  type: 'text',
  options: ['radio1', 'radio2']
}

// ❌ - We shouldn't be allowed to decalare placeholder on a Radio Choice
const myRadioChoice: Choice = {
  type: 'radio',
  placeholder: 'My placeholder'
}

✅ Discriminated unions

// MUCH BETTER
type TextChoice = {
  type: 'text'
  placeholder: string
}

type RadioChoice = {
  type: 'radio'
  options: string[]
}

// ✅
type Choice = TextChoice | RadioChoice

// Type '{ type: "text"; options: string[]; }' is not assignable to type 'Choice'.
const myTextChoice: Choice = {
  type: 'text',
  options: ['radio1', 'radio2']
}

Filtering undefined

❌ Using Boolean

// NOT SO CLEAN 
const filteredFruits = ['apple', 'orange', undefined, 'kiwi'].filter(Boolean)

[LOG]: ["apple", "orange", "kiwi"]
// const filteredFruits: (string | undefined)[]

✅ Type guard with type predicate

// MUCH BETTER
function isDefined<T>(listItem: T | undefined): listItem is T {
  if (listItem === undefined) {
    return false
  }
  return true
}

const filteredFruits = ['apple', 'orange', undefined, 'kiwi'].filter(isDefined)

// const filteredFruits: string[]

Returning Different Types

🤔

// NOT SO CLEAN 
// Contrived Example
type TypeA = string
type TypeB = number

function returnTypeAOrTypeB(shouldReturnTypeA: boolean) {
    if (shouldReturnTypeA) {
        return "23" as TypeA
    }
    return 23 as TypeB
}

const myValue = returnTypeAOrTypeB(true)   

// const myValue: string | number

✅ Function Overloading

// MUCH BETTER
type TypeA = string
type TypeB = number

function returnTypeAOrTypeB(shouldReturnTypeA: true): TypeA;

function returnTypeAOrTypeB(shouldReturnTypeA: false): TypeB;

function returnTypeAOrTypeB(shouldReturnTypeA: boolean) {
    if (shouldReturnTypeA) {
        return "23" as TypeA
    }
    return 23 as TypeB
}

const myValue = returnTypeAOrTypeB(true)

// const myValue: string

✅ Function Overloading with Conditional Types

// EVEN BETTER
type TypeA = string
type TypeB = number

function returnTypeAOrTypeB<T extends boolean>(shouldReturnTypeA: T): T extends true ? TypeA : TypeB

function returnTypeAOrTypeB(shouldReturnTypeA: boolean) {
    if (shouldReturnTypeA) {
        return "23" as TypeA
    }
    return 23 as TypeB
}

const myValue = returnTypeAOrTypeB(true) 

// const myValue: string

 Extract Keys of a Value Type

🤔

type Address = { 
  streetName: string;
  city: string;
  houseNumber: number;
  zipCode: number;
}

// type StringKeysOfAdresss = "streetName" | "city"

🙂 Types to extract keys

// WORKS BUT CAN BE CLEANER
type MatchingStringKeys<Type> = {
  [Key in keyof Type]: Type[Key] extends string ? Key : never
}[keyof Type]
 
// type StringKeysOfAdresss = "streetName" | "city"
type StringKeysOfAdresss = MatchingStringKeys<Address>

type MatchingNumberKeys<Type> = {
  [Key in keyof Type]: Type[Key] extends number ? Key : never
}[keyof Type]

// type NumberKeysOfAdresss = "houseNumber" | "zipCode"
type NumberKeysOfAdresss = MatchingNumberKeys<Address>

😁 Let's make it generic

// MUCH BETTER
type MatchingKeys<Type, FilterType> = {
  [Key in keyof Type]: Type[Key] extends FilterType
    ? Key
    : never
}[keyof Type]
 
// type StringKeysOfAdresss = "streetName" | "city"
type StringKeysOfAdresss = MatchingKeys<Address, string>

// type NumberKeysOfAdresss = "houseNumber" | "zipCode"
type NumberKeysOfAdresss = MatchingKeys<Address, number>
